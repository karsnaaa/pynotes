    """
    Classify the temperature in Celsius.

    Parameters:
    celsius (float): Temperature in degrees Celsius.

    Returns:
    str: Classification of the temperature.
    """
    if celsius < 10:
        return "Cold"
    elif 10 <= celsius <= 25:
        return "Moderate"
    else:
        return "Hot"
